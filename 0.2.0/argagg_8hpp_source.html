515 struct parser {
516 
522  std::vector<::argagg::definition> definitions;

529  ::argagg::parser_results parse(int argc, const char** argv)
530  {
531  bool ignore_flags = false;
532  const char* last_flag_expecting_args = nullptr;
533  option_result* last_option_expecting_args = nullptr;
534  char num_option_args_to_consume = 0;
535 
536  ::argagg::parser_results results { argv[0], {}, {} };
537 
538  // Add a result for each definition.
539  for (auto& defn : this->definitions) {
540  ::argagg::option_results opt_results {{}};
541  results.options.insert(std::make_pair(defn.name, opt_results));
542  }
543 
544  // Get pointers to pointers so we can treat the raw pointer array as an
545  // iterator for standard library algorithms.
546  const char** arg_i = argv + 1;
547  const char** arg_end = argv + argc;
548 
549  while (arg_i != arg_end) {
550  auto arg_i_str = *arg_i;
551 
552  // If we're not ignoring flags then check to see if this argument looks
553  // like a definition that has been specified...
554  auto matching_flag = this->definitions.end();
555  if (!ignore_flags) {
556  matching_flag = std::find_if(
557  this->definitions.begin(), this->definitions.end(),
558  [=](const ::argagg::definition& defn) {
559  // ...and it looks like a definition that has been specified if the
560  // argument matches "any of" the flag strings for a given
561  // definition.
562  return std::any_of(
563  defn.flags.begin(), defn.flags.end(),
564  [=](const char* flag) {
565  return std::strcmp(flag, arg_i_str) == 0;
566  });
567  });
568  }
569 
570  if (matching_flag == this->definitions.end()) {
571  // If we are ignoring flags or the argument didn't match a definition
572  // that was specified...
573 
574  if (!ignore_flags && arg_i_str[0] == '-') {
575  // If this argument actually looks like a flag...
576 
577  if (arg_i_str[1] == '-' && arg_i_str[2] == '\0') {
578  // If the argument is "--" then that means treat the rest of the
579  // arguments as positional arguments.
580  ignore_flags = true;
581 
582  } else {
583  // If we get a option and it isn't one that was specified nor "--"
584  // then this is an error.
585  std::ostringstream msg;
586  msg << "found unexpected flag: " << arg_i_str;
587  throw unexpected_option_error(msg.str());
588  }
589 
590  } else if (num_option_args_to_consume > 0) {
591  // If we get an argument and the last option is expecting some
592  // specific positive number of arguments then give this positional
593  // argument to that option.
594  last_option_expecting_args->arg = arg_i_str;
595  --num_option_args_to_consume;
596 
597  } else if (num_option_args_to_consume == ::argagg::optional) {
598  // If we get an argument and the last option expects zero or one
599  // arguments then give this positional argument to the option and
600  // change mode so that we aren't expecting anymore option arguments.
601  last_option_expecting_args->arg = arg_i_str;
602  num_option_args_to_consume = 0;
603 
604  } else {
605  // If there are no expectations for option arguments then simply use
606  // this argument as a positional argument.
607  results.pos.push_back(arg_i_str);
608 
609  }
610  } else {
611  // If we found a option that was specified...
612 
613  if (num_option_args_to_consume > 0) {
614  // If we get a known option but are expecting an argument for a
615  // previous option then throw an error.
616  std::ostringstream msg;
617  msg << "option \"" << last_flag_expecting_args
618  << "\" expects an argument but got another option (\""
619  << arg_i_str << "\") instead";
620  throw option_lacks_argument_error(msg.str());
621 
622  } else if (num_option_args_to_consume == ::argagg::optional) {
623  // If we get a known option but are expecting zero or one arguments
624  // for the last option then simply reset the number of expected
625  // option arguments.
626  num_option_args_to_consume = 0;
627  }
628 
629  // Get the option results.
630  auto& defn = *matching_flag;
631  auto& opt_results = results.options[defn.name];
632 
633  // Add a single option result to option results.
634  ::argagg::option_result opt_result {nullptr};
635  opt_results.all.push_back(std::move(opt_result));
636 
637  // If this option expects arguments then put our parser into the
638  // corresponding "expecting option arguments" mode.
639  if (defn.num_args != 0) {
640  last_flag_expecting_args = arg_i_str;
641  last_option_expecting_args = &(opt_results.all.back());
642  num_option_args_to_consume = defn.num_args;
643  } else {
644  last_flag_expecting_args = nullptr;
645  last_option_expecting_args = nullptr;
646  num_option_args_to_consume = 0;
647  }
648  }
649 
650  // Go on to the next argument.
651  ++arg_i;
652  }
653 
654  // If we're done with all of the arguments but are still expecting
655  // arguments for a previous option then we haven't satisfied that option.
656  // This is an error.
657  if (num_option_args_to_consume > 0) {
658  std::ostringstream msg;
659  msg << "last option \"" << last_flag_expecting_args
660  << "\" expects an argument but the parser ran out of command line "
661  << "arguments to parse";
662  throw option_lacks_argument_error(msg.str());
663  }
664 
665  return results;
666  }
