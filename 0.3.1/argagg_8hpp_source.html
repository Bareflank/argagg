558 bool cmd_line_arg_is_option_flag(
559  const char* s)
560 {
561  auto len = std::strlen(s);
562 
563  // The shortest possible flag has two characters: a hyphen and an
564  // alpha-numeric character.
565  if (len < 2) {
566  return false;
567  }
568 
569  // All flags must start with a hyphen.
570  if (s[0] != '-') {
571  return false;
572  }
573 
574  // Shift the name forward by a character to account for the initial hyphen.
575  // This means if s was originally "-v" then name will be "v".
576  const char* name = s + 1;
577 
578  // Check if we're dealing with a long flag.
579  bool is_long = false;
580  if (s[1] == '-') {
581  is_long = true;
582 
583  // Just -- is not a valid flag.
584  if (len == 2) {
585  return false;
586  }
587 
588  // Shift the name forward to account for the extra hyphen. This means if s
589  // was originally "--output" then name will be "output".
590  name = s + 2;
591  }
592 
593  // The first character of the flag name must be alpha-numeric. This is to
594  // prevent things like "---a" from being valid flags.
595  len = std::strlen(name);
596  if (!std::isalnum(name[0])) {
597  return false;
598  }
599 
600  // At this point in is_valid_flag_definition() we would check if the short
601  // flag has only one character. At command line specification you can group
602  // short flags together or even add an argument to a short flag without a
603  // space delimiter. Thus we don't check if this has only one character
604  // because it might not.
605 
606  // If this is a long flag then we expect all characters *up to* an equal sign
607  // to be alpha-numeric or a hyphen. After the equal sign you are specify the
608  // argument to a long flag which can be basically anything.
609  if (is_long) {
610  bool encountered_equal = false;
611  return std::all_of(name, name + len, [&](const char& c) {
612  if (encountered_equal) {
613  return true;
614  } else {
615  if (c == '=') {
616  encountered_equal = true;
617  return true;
618  }
619  return std::isalnum(c) || c == '-';
620  }
621  });
622  }
623 
624  // At this point we are not dealing with a long flag. We already checked that
625  // the first character is alpha-numeric so we've got the case of a single
626  // short flag covered. This might be a short flag group though and we might
627  // be tempted to check that each character of the short flag group is
628  // alpha-numeric. However, you can specify the argument for a short flag
629  // without a space delimiter (e.g. "-I/usr/local/include") so you can't tell
630  // if the rest of a short flag group is part of the argument or not unless
631  // you know what is a defined flag or not. We leave that kind of processing
632  // to the parser.
633  return true;
634 }

643  const char* s)

654  if (s[0] != '-') {
655  return false;
656  }
657

659  // This means if s was originally "-v" then name will be "v".
660  const char* name = s + 1;
661 
662  // Check if we're dealing with a long flag.
663  bool is_long = false;
664  if (s[1] == '-') {
665  is_long = true;
666 
667  // Just -- is not a valid flag.
668  if (len == 2) {
669  return false;
670  }
671 
672  // Shift the name forward to account for the extra hyphen. This means if s
673  // was originally "--output" then name will be "output".
674  name = s + 2;
675  }
676 
677  // The first character of the flag name must be alpha-numeric. This is to
678  // prevent things like "---a" from being valid flags.
679  len = std::strlen(name);
680  if (!std::isalnum(name[0])) {
681  return false;
682  }
683 
684  // If this is a short flag then it must only have one character.
685  if (!is_long && len > 1) {
686  return false;
687  }
688 
689  // The rest of the characters must be alpha-numeric, but long flags are
690  // allowed to have hyphens too.
691  return std::all_of(name + 1, name + len, [&](const char& c) {
692  return std::isalnum(c) || (c == '-' && is_long);
693  });
694 }

793 parser_map validate_definitions(
794  const std::vector<definition>& definitions)
795 {
796  parser_map map {{nullptr}, {}};
797 
798  for (auto& defn : definitions) {
799 
800  if (defn.flags.size() == 0) {
801  std::ostringstream msg;
802  msg << "option \"" << defn.name << "\" has no flag definitions";
803  throw invalid_flag(msg.str());
804  }
805 
806  for (auto& flag : defn.flags) {
807 
808  if (!is_valid_flag_definition(flag)) {
809  std::ostringstream msg;
810  msg << "flag \"" << flag << "\" specified for option \"" << defn.name
811  << "\" is invalid";
812  throw invalid_flag(msg.str());
813  }
814 
815  if (flag_is_short(flag)) {
816  const int short_flag_letter = flag[1];
817  const auto existing_short_flag = map.short_map[short_flag_letter];
818  bool short_flag_already_exists = (existing_short_flag != nullptr);
819  if (short_flag_already_exists) {
820  std::ostringstream msg;
821  msg << "duplicate short flag \"" << flag
822  << "\" found, specified by both option \"" << defn.name
823  << "\" and option \"" << existing_short_flag->name;
824  throw invalid_flag(msg.str());
825  }
826  map.short_map[short_flag_letter] = &defn;
827  continue;
828  }
829 
830  // If we're here then this is a valid, long-style flag.
831  if (map.known_long_flag(flag)) {
832  const auto existing_long_flag = map.get_definition_for_long_flag(flag);
833  std::ostringstream msg;
834  msg << "duplicate long flag \"" << flag
835  << "\" found, specified by both option \"" << defn.name
836  << "\" and option \"" << existing_long_flag->name;
837  throw invalid_flag(msg.str());
838  }
839  map.long_map.insert(std::make_pair(flag, &defn));
840  }
841  }
842 
843  return map;
844 }

859 
869  parser_results parse(int argc, const char** argv) const
870  {
871  // Inspect each definition to see if its valid. You may wonder "why don't
872  // you do this validation on construction?" I had thought about it but
873  // realized that since I've made the parser an aggregate type (granted it
874  // just "aggregates" a single vector) I would need to track any changes to
875  // the definitions vector and re-run the validity check in order to
876  // maintain this expected "validity invariant" on the object. That would
877  // then require hiding the definitions vector as a private entry and then
878  // turning the parser into a thin interface (by re-exposing setters and
879  // getters) to the vector methods just so that I can catch when the
880  // definition has been modified. It seems much simpler to just enforce the
881  // validity when you actually want to parser because it's at the moment of
882  // parsing that you know the definitions are complete.
883  parser_map map = validate_definitions(this->definitions);
884 
885  // Initialize the parser results that we'll be returning. Store the program
886  // name (assumed to be the first command line argument) and initialize
887  // everything else as empty.
888  std::unordered_map<std::string, option_results> options {};
889  parser_results results {argv[0], std::move(options), {}};
890 
891  // Add an empty option result for each definition.
892  for (const auto& defn : this->definitions) {
893  option_results opt_results {{}};
894  results.options.insert(
895  std::move(std::make_pair(defn.name, opt_results)));
896  }
897 
898  // Don't start off ignoring flags. We only ignore flags after a -- shows up
899  // in the command line arguments.
900  bool ignore_flags = false;
901 
902  // Keep track of any options that are expecting arguments.
903  const char* last_flag_expecting_args = nullptr;
904  option_result* last_option_expecting_args = nullptr;
905  unsigned int num_option_args_to_consume = 0;
906 
907  // Get pointers to pointers so we can treat the raw pointer array as an
908  // iterator for standard library algorithms. This isn't used yet but can be
909  // used to template this function to work on iterators over strings or
910  // C-strings.
911  const char** arg_i = argv + 1;
912  const char** arg_end = argv + argc;
913 
914  while (arg_i != arg_end) {
915  auto arg_i_cstr = *arg_i;
916  auto arg_i_len = std::strlen(arg_i_cstr);
917 
918  // Some behavior to note: if the previous option is expecting an argument
919  // then the next entry will be treated as a positional argument even if
920  // it looks like a flag.
921  bool treat_as_positional_argument = (
922  ignore_flags
923  || num_option_args_to_consume > 0
924  || !cmd_line_arg_is_option_flag(arg_i_cstr)
925  );
926  if (treat_as_positional_argument) {
927 
928  // If last option is expecting some specific positive number of
929  // arguments then give this argument to that option, *regardless of
930  // whether or not the argument looks like a flag or is the special "--"
931  // argument*.
932  if (num_option_args_to_consume > 0) {
933  last_option_expecting_args->arg = arg_i_cstr;
934  --num_option_args_to_consume;
935  ++arg_i;
936  continue;
937  }
938 
939  // Now we check if this is just "--" which is a special argument that
940  // causes all following arguments to be treated as non-options and is
941  // itselve discarded.
942  if (std::strncmp(arg_i_cstr, "--", 2) == 0 && arg_i_len == 2) {
943  ignore_flags = true;
944  ++arg_i;
945  continue;
946  }
947 
948  // If there are no expectations for option arguments then simply use
949  // this argument as a positional argument.
950  results.pos.push_back(arg_i_cstr);
951  ++arg_i;
952  continue;
953  }
954 
955  // Reset the "expecting argument" state.
956  last_flag_expecting_args = nullptr;
957  last_option_expecting_args = nullptr;
958  num_option_args_to_consume = 0;
959 
960  // If we're at this point then we're definitely dealing with something
961  // that is flag-like and has hyphen as the first character and has a
962  // length of at least two characters. How we handle this potential flag
963  // depends on whether or not it is a long-option so we check that first.
964  bool is_long_flag = (arg_i_cstr[1] == '-');
965 
966  if (is_long_flag) {
967 
968  // Long flags have a complication: their arguments can be specified
969  // using an '=' character right inside the argument. That means an
970  // argument like "--output=foobar.txt" is actually an option with flag
971  // "--output" and argument "foobar.txt". So we look for the first
972  // instance of the '=' character and keep it in long_flag_arg. If
973  // long_flag_arg is nullptr then we didn't find '='. We need the
974  // flag_len to construct long_flag_str below.
975  auto long_flag_arg = std::strchr(arg_i_cstr, '=');
976  std::size_t flag_len = arg_i_len;
977  if (long_flag_arg != nullptr) {
978  flag_len = long_flag_arg - arg_i_cstr;
979  }
980  std::string long_flag_str(arg_i_cstr, flag_len);
981 
982  if (!map.known_long_flag(long_flag_str)) {
983  std::ostringstream msg;
984  msg << "found unexpected flag: " << long_flag_str;
985  throw unexpected_option_error(msg.str());
986  }
987 
988  const auto defn = map.get_definition_for_long_flag(long_flag_str);
989 
990  if (long_flag_arg != nullptr && defn->num_args == 0) {
991  std::ostringstream msg;
992  msg << "found argument for option not expecting an argument: "
993  << arg_i_cstr;
994  throw unexpected_argument_error(msg.str());
995  }
996 
997  // We've got a legitimate, known long flag option so we add an option
998  // result. This option result initially has an arg of nullptr, but that
999  // might change in the following block.
1000  auto& opt_results = results.options[defn->name];
1001  option_result opt_result {nullptr};
1002  opt_results.all.push_back(std::move(opt_result));
1003 
1004  if (defn->requires_arguments()) {
1005  bool there_is_an_equal_delimited_arg = (long_flag_arg != nullptr);
1006  if (there_is_an_equal_delimited_arg) {
1007  // long_flag_arg would be "=foo" in the "--output=foo" case so we
1008  // increment by 1 to get rid of the equal sign.
1009  opt_results.all.back().arg = long_flag_arg + 1;
1010  } else {
1011  last_flag_expecting_args = arg_i_cstr;
1012  last_option_expecting_args = &(opt_results.all.back());
1013  num_option_args_to_consume = defn->num_args;
1014  }
1015  }
1016 
1017  ++arg_i;
1018  continue;
1019  }
1020 
1021  // If we've made it here then we're looking at either a short flag or a
1022  // group of short flags. Short flags can be grouped together so long as
1023  // they don't require any arguments unless the option that does is the
1024  // last in the group ("-o x -v" is okay, "-vo x" is okay, "-ov x" is
1025  // not). So starting after the dash we're going to process each character
1026  // as if it were a separate flag. Note "sf_idx" stands for "short flag
1027  // index".
1028  for (std::size_t sf_idx = 1; sf_idx < arg_i_len; ++sf_idx) {
1029  const auto short_flag = arg_i_cstr[sf_idx];
1030 
1031  if (!std::isalnum(short_flag)) {
1032  std::ostringstream msg;
1033  msg << "found non-alphanumeric character '" << arg_i_cstr[sf_idx]
1034  << "' in flag group '" << arg_i_cstr << "'";
1035  throw std::domain_error(msg.str());
1036  }
1037 
1038  if (!map.known_short_flag(short_flag)) {
1039  std::ostringstream msg;
1040  msg << "found unexpected flag '" << arg_i_cstr[sf_idx]
1041  << "' in flag group '" << arg_i_cstr << "'";
1042  throw unexpected_option_error(msg.str());
1043  }
1044 
1045  auto defn = map.get_definition_for_short_flag(short_flag);
1046  auto& opt_results = results.options[defn->name];
1047 
1048  // Create an option result with an empty argument (for now) and add it
1049  // to this option's results.
1050  option_result opt_result {nullptr};
1051  opt_results.all.push_back(std::move(opt_result));
1052 
1053  if (defn->requires_arguments()) {
1054 
1055  // If this short flag's option requires an argument and we're the
1056  // last flag in the short flag group then just put the parser into
1057  // "expecting argument for last option" state and move onto the next
1058  // command line argument.
1059  bool is_last_short_flag_in_group = (sf_idx == arg_i_len - 1);
1060  if (is_last_short_flag_in_group) {
1061  last_flag_expecting_args = arg_i_cstr;
1062  last_option_expecting_args = &(opt_results.all.back());
1063  num_option_args_to_consume = defn->num_args;
1064  break;
1065  }
1066 
1067  // If this short flag's option requires an argument and we're NOT the
1068  // last flag in the short flag group then we automatically consume
1069  // the rest of the short flag group as the argument for this flag.
1070  // This is how we get the POSIX behavior of being able to specify a
1071  // flag's arguments without a white space delimiter (e.g.
1072  // "-I/usr/local/include").
1073  opt_results.all.back().arg = arg_i_cstr + sf_idx + 1;
1074  break;
1075  }
1076  }
1077 
1078  ++arg_i;
1079  continue;
1080  }
1081 
1082  // If we're done with all of the arguments but are still expecting
