307 struct parser {
308 
314  args parse(int argc, const char** argv)
315  {
316  bool ignore_flags = false;
317  flag* last_flag_expecting_args = nullptr;
318  int num_flag_args_to_consume = 0;
319 
320  args arg_res { argv[0], {}, {} };
321 
322  const char** arg_i = argv + 1;
323  const char** arg_end = argv + argc;
324 
325  while (arg_i != arg_end) {
326 
327  // If we're not ignoring flags then check to see if this argument looks
328  // like a flag_spec that has been specified...
329  auto matching_flag = ignore_flags ? specs.end() :
330  std::find_if(
331  this->specs.begin(), this->specs.end(),
332  [=](const flag_spec& spec) {
333  // ...and it looks like a flag_spec that has been specified if the
334  // argument matches "any of" the flag strings for a given
335  // flag_spec.
336  return std::any_of(
337  spec.flags.begin(), spec.flags.end(),
338  [=](const char* flag) {
339  return std::strcmp(flag, *arg_i) == 0;
340  });
341  });
342 
343  if (matching_flag == specs.end()) {
344  // If we are ignoring flags or the argument didn't match a flag_spec
345  // that was specified...
346 
347  auto arg_i_str = *arg_i;
348  if (!ignore_flags && arg_i_str[0] == '-') {
349  // If this argument actually looks like a flag...
350 
351  if (arg_i_str[1] == '-' && arg_i_str[2] == '\0') {
352  // If the argument is "--" then that means treat the rest of the
353  // arguments are positional arguments.
354  ignore_flags = true;
355 
356  } else {
357  // If we get a flag and it isn't one that was specified nor "--"
358  // then this is an error.
359  throw unexpected_flag_error("found unused flag");
360  }
361 
362  } else if (num_flag_args_to_consume > 0) {
363  // If we get an argument and the last flag is expecting some specific
364  // positive number of arguments then give this positional argument to
365  // that flag.
366  last_flag_expecting_args->args.push_back(*arg_i);
367  --num_flag_args_to_consume;
368 
369  } else if (num_flag_args_to_consume == flag_spec::zero_plus) {
370  // If we get an argument and the last flag wants as many (or zero)
371  // arguments as it can get then give this positional argument to that
372  // flag and stay in this mode.
373  last_flag_expecting_args->args.push_back(*arg_i);
374 
375  } else if (num_flag_args_to_consume == flag_spec::one_plus) {
376  // If we get an argument and the last flag wants as many (but at
377  // least one) arguments as it can get then give this positional
378  // argument to that flag but go into the mode where we was as many or
379  // zero arguments because that "at least one" we wanted has been
380  // satisfied.
381  last_flag_expecting_args->args.push_back(*arg_i);
382  num_flag_args_to_consume = flag_spec::zero_plus;
383 
384  } else if (num_flag_args_to_consume == flag_spec::optional) {
385  // If we get an argument and the last flag expects zero or one
386  // arguments then give this positional argument to the flag and
387  // change mode so that we aren't expecting anymore flag arguments.
388  last_flag_expecting_args->args.push_back(*arg_i);
389  num_flag_args_to_consume = 0;
390 
391  } else {
392  // If there are no expectations for flag arguments then simply use
393  // this argument as a positional argument.
394  arg_res.args.push_back(*arg_i);
395 
396  }
397  } else {
398  // If we found a flag that was specified...
399 
400  if (num_flag_args_to_consume > 0) {
401  // If we get a known flag but are expecting arguments for a previous
402  // flag then throw an error.
403  throw unexpected_flag_error(
404  "expected arguments for previous flag but got another flag");
405 
406  } else if (num_flag_args_to_consume == flag_spec::zero_plus) {
407  // If we get a known flag but are expecting as many (or zero)
408  // arguments for the last flag then simply reset the number of
409  // expected flag arguments.
410  num_flag_args_to_consume = 0;
411 
412  } else if (num_flag_args_to_consume == flag_spec::one_plus) {
413  // If we get a known flag but are still expecting at least one or
414  // more arguments for the last flag then we haven't found "at least
415  // one" argument for the last flag. This is an error.
416  throw unexpected_flag_error(
417  "expected arguments for previous flag but got another flag");
418 
419  } else if (num_flag_args_to_consume == flag_spec::optional) {
420  // If we get a known flag but are expecting zero or one arguments for
421  // the last flag then simply reset the number of expected flag
422  // arguments.
423  num_flag_args_to_consume = 0;
424  }
425 
426  auto& spec = *matching_flag;
427 
428  // If this flag_spec has not been added before then do that now.
429  if (!arg_res.has_flag(spec.name)) {
430  flag new_flag {0, {}};
431  arg_res.flags.emplace(
432  std::make_pair(spec.name, std::move(new_flag)));
433  }
434 
435  flag& flag_i = arg_res.flags[spec.name];
436 
437  // Increment the number of occurences of this flag
438  ++flag_i.occurences;
439 
440  // If this flag expects arguments then put our parser into the
441  // corresponding "expected flag arguments" mode.
442  if (spec.num_args != 0) {
443  last_flag_expecting_args = &flag_i;
444  num_flag_args_to_consume = spec.num_args;
445  }
446  }
447 
448  // Go on to the next argument.
449  ++arg_i;
450  }
451 
452  // If we're done with all of the arguments but are still expecting
453  // arguments for a previous flag then we haven't satisfied that flag. This
454  // is an error.
455  if (num_flag_args_to_consume > 0 ||
456  num_flag_args_to_consume == flag_spec::one_plus) {
457  throw std::out_of_range(
458  "flag expected arguments but there are no more arguments");
459  }
460 
461  return arg_res;
462  }
