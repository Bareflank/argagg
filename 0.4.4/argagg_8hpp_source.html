877 inline
878 bool cmd_line_arg_is_option_flag(
879  const char* s)
880 {
881  auto len = std::strlen(s);
882 
883  // The shortest possible flag has two characters: a hyphen and an
884  // alpha-numeric character.
885  if (len < 2) {
886  return false;
887  }
888 
889  // All flags must start with a hyphen.
890  if (s[0] != '-') {
891  return false;
892  }
893 
894  // Shift the name forward by a character to account for the initial hyphen.
895  // This means if s was originally "-v" then name will be "v".
896  const char* name = s + 1;
897 
898  // Check if we're dealing with a long flag.
899  bool is_long = false;
900  if (s[1] == '-') {
901  is_long = true;
902 
903  // Just -- is not a valid flag.
904  if (len == 2) {
905  return false;
906  }
907 
908  // Shift the name forward to account for the extra hyphen. This means if s
909  // was originally "--output" then name will be "output".
910  name = s + 2;
911  }
912 
913  // The first character of the flag name must be alpha-numeric. This is to
914  // prevent things like "---a" from being valid flags.
915  len = std::strlen(name);
916  if (!std::isalnum(name[0])) {
917  return false;
918  }
919 
920  // At this point in is_valid_flag_definition() we would check if the short
921  // flag has only one character. At command line specification you can group
922  // short flags together or even add an argument to a short flag without a
923  // space delimiter. Thus we don't check if this has only one character
924  // because it might not.
925 
926  // If this is a long flag then we expect all characters *up to* an equal sign
927  // to be alpha-numeric or a hyphen. After the equal sign you are specify the
928  // argument to a long flag which can be basically anything.
929  if (is_long) {
930  bool encountered_equal = false;
931  return std::all_of(name, name + len, [&](const char& c) {
932  if (encountered_equal) {
933  return true;
934  } else {
935  if (c == '=') {
936  encountered_equal = true;
937  return true;
938  }
939  return std::isalnum(c) || c == '-';
940  }
941  });
942  }
943 
944  // At this point we are not dealing with a long flag. We already checked that
945  // the first character is alpha-numeric so we've got the case of a single
946  // short flag covered. This might be a short flag group though and we might
947  // be tempted to check that each character of the short flag group is
948  // alpha-numeric. However, you can specify the argument for a short flag
949  // without a space delimiter (e.g. "-I/usr/local/include") so you can't tell
950  // if the rest of a short flag group is part of the argument or not unless
951  // you know what is a defined flag or not. We leave that kind of processing
952  // to the parser.
953  return true;
954 }

957 inline
958 bool is_valid_flag_definition(
959  const char* s)
960 {
961  auto len = std::strlen(s);
962 
963  // The shortest possible flag has two characters: a hyphen and an
964  // alpha-numeric character.
965  if (len < 2) {
966  return false;
967  }
968 
969  // All flags must start with a hyphen.
970  if (s[0] != '-') {
971  return false;
972  }
973 
974  // Shift the name forward by a character to account for the initial hyphen.
975  // This means if s was originally "-v" then name will be "v".
976  const char* name = s + 1;
977 
978  // Check if we're dealing with a long flag.
979  bool is_long = false;
980  if (s[1] == '-') {
981  is_long = true;
982 
983  // Just -- is not a valid flag.
984  if (len == 2) {
985  return false;
986  }
987 
988  // Shift the name forward to account for the extra hyphen. This means if s
989  // was originally "--output" then name will be "output".
990  name = s + 2;
991  }
992 
993  // The first character of the flag name must be alpha-numeric. This is to
994  // prevent things like "---a" from being valid flags.
995  len = std::strlen(name);
996  if (!std::isalnum(name[0])) {
997  return false;
998  }
999 
1000  // If this is a short flag then it must only have one character.
1001  if (!is_long && len > 1) {
1002  return false;
1003  }
1004 
1005  // The rest of the characters must be alpha-numeric, but long flags are
1006  // allowed to have hyphens too.
1007  return std::all_of(name + 1, name + len, [&](const char& c) {
1008  return std::isalnum(c) || (c == '-' && is_long);
1009  });
1010 }

1058 inline
1059 parser_map validate_definitions(
1060  const std::vector<definition>& definitions)
1061 {
1062  std::unordered_map<std::string, const definition*> long_map;
1063  parser_map map {{{nullptr}}, std::move(long_map)};
1064 
1065  for (auto& defn : definitions) {
1066 
1067  if (defn.flags.size() == 0) {
1068  std::ostringstream msg;
1069  msg << "option \"" << defn.name << "\" has no flag definitions";
1070  throw invalid_flag(msg.str());
1071  }
1072 
1073  for (auto& flag : defn.flags) {
1074 
1075  if (!is_valid_flag_definition(flag.data())) {
1076  std::ostringstream msg;
1077  msg << "flag \"" << flag << "\" specified for option \"" << defn.name
1078  << "\" is invalid";
1079  throw invalid_flag(msg.str());
1080  }
1081 
1082  if (flag_is_short(flag.data())) {
1083  const int short_flag_letter = flag[1];
1084  const auto existing_short_flag = map.short_map[short_flag_letter];
1085  bool short_flag_already_exists = (existing_short_flag != nullptr);
1086  if (short_flag_already_exists) {
1087  std::ostringstream msg;
1088  msg << "duplicate short flag \"" << flag
1089  << "\" found, specified by both option \"" << defn.name
1090  << "\" and option \"" << existing_short_flag->name;
1091  throw invalid_flag(msg.str());
1092  }
1093  map.short_map[short_flag_letter] = &defn;
1094  continue;
1095  }
1096 
1097  // If we're here then this is a valid, long-style flag.
1098  if (map.known_long_flag(flag)) {
1099  const auto existing_long_flag = map.get_definition_for_long_flag(flag);
1100  std::ostringstream msg;
1101  msg << "duplicate long flag \"" << flag
1102  << "\" found, specified by both option \"" << defn.name
1103  << "\" and option \"" << existing_long_flag->name;
1104  throw invalid_flag(msg.str());
1105  }
1106  map.long_map.insert(std::make_pair(flag, &defn));
1107  }
1108  }
1109 
1110  return map;
1111 }

1114 inline
1115 parser_results parser::parse(int argc, const char** argv) const
1116 {
1117  // Inspect each definition to see if its valid. You may wonder "why don't
1118  // you do this validation on construction?" I had thought about it but
1119  // realized that since I've made the parser an aggregate type (granted it
1120  // just "aggregates" a single vector) I would need to track any changes to
1121  // the definitions vector and re-run the validity check in order to
1122  // maintain this expected "validity invariant" on the object. That would
1123  // then require hiding the definitions vector as a private entry and then
1124  // turning the parser into a thin interface (by re-exposing setters and
1125  // getters) to the vector methods just so that I can catch when the
1126  // definition has been modified. It seems much simpler to just enforce the
1127  // validity when you actually want to parser because it's at the moment of
1128  // parsing that you know the definitions are complete.
1129  parser_map map = validate_definitions(this->definitions);
1130 
1131  // Initialize the parser results that we'll be returning. Store the program
1132  // name (assumed to be the first command line argument) and initialize
1133  // everything else as empty.
1134  std::unordered_map<std::string, option_results> options {};
1135  std::vector<const char*> pos;
1136  parser_results results {argv[0], std::move(options), std::move(pos)};
1137 
1138  // Add an empty option result for each definition.
1139  for (const auto& defn : this->definitions) {
1140  option_results opt_results {{}};
1141  results.options.insert(
1142  std::make_pair(defn.name, opt_results));
1143  }
1144 
1145  // Don't start off ignoring flags. We only ignore flags after a -- shows up
1146  // in the command line arguments.
1147  bool ignore_flags = false;
1148 
1149  // Keep track of any options that are expecting arguments.
1150  const char* last_flag_expecting_args = nullptr;
1151  option_result* last_option_expecting_args = nullptr;
1152  unsigned int num_option_args_to_consume = 0;
1153 
1154  // Get pointers to pointers so we can treat the raw pointer array as an
1155  // iterator for standard library algorithms. This isn't used yet but can be
1156  // used to template this function to work on iterators over strings or
1157  // C-strings.
1158  const char** arg_i = argv + 1;
1159  const char** arg_end = argv + argc;
1160 
1161  while (arg_i != arg_end) {
1162  auto arg_i_cstr = *arg_i;
1163  auto arg_i_len = std::strlen(arg_i_cstr);
1164 
1165  // Some behavior to note: if the previous option is expecting an argument
1166  // then the next entry will be treated as a positional argument even if
1167  // it looks like a flag.
1168  bool treat_as_positional_argument = (
1169  ignore_flags
1170  || num_option_args_to_consume > 0
1171  || !cmd_line_arg_is_option_flag(arg_i_cstr)
1172  );
1173  if (treat_as_positional_argument) {
1174 
1175  // If last option is expecting some specific positive number of
1176  // arguments then give this argument to that option, *regardless of
1177  // whether or not the argument looks like a flag or is the special "--"
1178  // argument*.
1179  if (num_option_args_to_consume > 0) {
1180  last_option_expecting_args->arg = arg_i_cstr;
1181  --num_option_args_to_consume;
1182  ++arg_i;
1183  continue;
1184  }
1185 
1186  // Now we check if this is just "--" which is a special argument that
1187  // causes all following arguments to be treated as non-options and is
1188  // itselve discarded.
1189  if (std::strncmp(arg_i_cstr, "--", 2) == 0 && arg_i_len == 2) {
1190  ignore_flags = true;
1191  ++arg_i;
1192  continue;
1193  }
1194 
1195  // If there are no expectations for option arguments then simply use
1196  // this argument as a positional argument.
1197  results.pos.push_back(arg_i_cstr);
1198  ++arg_i;
1199  continue;
1200  }
1201 
1202  // Reset the "expecting argument" state.
1203  last_flag_expecting_args = nullptr;
1204  last_option_expecting_args = nullptr;
1205  num_option_args_to_consume = 0;
1206 
1207  // If we're at this point then we're definitely dealing with something
1208  // that is flag-like and has hyphen as the first character and has a
1209  // length of at least two characters. How we handle this potential flag
1210  // depends on whether or not it is a long-option so we check that first.
1211  bool is_long_flag = (arg_i_cstr[1] == '-');
1212 
1213  if (is_long_flag) {
1214 
1215  // Long flags have a complication: their arguments can be specified
1216  // using an '=' character right inside the argument. That means an
1217  // argument like "--output=foobar.txt" is actually an option with flag
1218  // "--output" and argument "foobar.txt". So we look for the first
1219  // instance of the '=' character and keep it in long_flag_arg. If
1220  // long_flag_arg is nullptr then we didn't find '='. We need the
1221  // flag_len to construct long_flag_str below.

1241  throw unexpected_argument_error(msg.str());

1246  // might change in the following block.

1288  << "' in flag group '" << arg_i_cstr << "'";
1289  throw unexpected_option_error(msg.str());
1290  }

1293  auto& opt_results = results.options[defn->name];

1296  // to this option's results.
1297  option_result opt_result {nullptr};
1298  opt_results.all.push_back(std::move(opt_result));

1303  // last flag in the short flag group then just put the parser into
1304  // "expecting argument for last option" state and move onto the next
1305  // command line argument.
1306  bool is_last_short_flag_in_group = (sf_idx == arg_i_len - 1);
1307  if (is_last_short_flag_in_group) {
1308  last_flag_expecting_args = arg_i_cstr;
1309  last_option_expecting_args = &(opt_results.all.back());
1310  num_option_args_to_consume = defn->num_args;
1311  break;
1312  }

1315  // last flag in the short flag group then we automatically consume
1316  // the rest of the short flag group as the argument for this flag.
1317  // This is how we get the POSIX behavior of being able to specify a
1318  // flag's arguments without a white space delimiter (e.g.
1319  // "-I/usr/local/include").
1320  opt_results.all.back().arg = arg_i_cstr + sf_idx + 1;
1321  break;
1322  }
1323  }
1324 
1325  ++arg_i;
1326  continue;
1327  }
1328

1330  // arguments for a previous option then we haven't satisfied that option.
1331  // This is an error.
1332  if (num_option_args_to_consume > 0) {
1333  std::ostringstream msg;
1334  msg << "last option \"" << last_flag_expecting_args
1335  << "\" expects an argument but the parser ran out of command line "
1336  << "arguments to parse";
1337  throw option_lacks_argument_error(msg.str());
1338  }
1339 
1340  return results;
1341 }
